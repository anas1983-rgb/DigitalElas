package com.ana.digitalelas

import android.content.Intent
import android.os.Bundle
import android.speech.tts.TextToSpeech
import android.widget.Button
import android.widget.ImageButton
import android.widget.Toast
import androidx.appcompat.app.AppCompatActivity
import java.util.Locale

class MainActivity : AppCompatActivity(), TextToSpeech.OnInitListener {

    private lateinit var tts: TextToSpeech
    private var idiomaSelecionado: String = "pt"

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        setContentView(R.layout.activity_main)

        // Inicializar Text-to-Speech
        tts = TextToSpeech(this, this)

        // Configurar botões
        configurarBotoes()
    }

    private fun configurarBotoes() {
        val btnPortugues = findViewById<Button>(R.id.btnPortugues)
        val btnEspanhol = findViewById<Button>(R.id.btnEspanhol)
        val btnOuvirOpcoes = findViewById<ImageButton>(R.id.btnOuvirOpcoes)

        // Botão Português
        btnPortugues.setOnClickListener {
            idiomaSelecionado = "pt"
            Toast.makeText(this, "Português selecionado!", Toast.LENGTH_SHORT).show()
            falarTexto("Português selecionado. Bem-vinda ao Digital Elas!")

            val intent = Intent(this, MenuModulosActivity::class.java)
            intent.putExtra("IDIOMA", "pt")
            startActivity(intent)
        }

        // Botão Espanhol
        btnEspanhol.setOnClickListener {
            idiomaSelecionado = "es"
            Toast.makeText(this, "¡Español seleccionado!", Toast.LENGTH_SHORT).show()
            falarTexto("Español seleccionado. ¡Bienvenida a Digital Elas!")

            // ADICIONE ESTAS 2 LINHAS ↓
            val intent = Intent(this, MenuModulosActivity::class.java)
            intent.putExtra("IDIOMA", "es")
            startActivity(intent)

        }

        // Botão Ouvir Opções
        btnOuvirOpcoes.setOnClickListener {
            when (idiomaSelecionado) {
                "pt" -> falarTexto("Escolha seu idioma: Português ou Espanhol")
                "es" -> falarTexto("Elija su idioma: Portugués o Español")
            }
        }
    }

    private fun falarTexto(texto: String) {
        when (idiomaSelecionado) {
            "pt" -> tts.language = Locale("pt", "BR")
            "es" -> tts.language = Locale("es", "ES")
        }

        tts.speak(texto, TextToSpeech.QUEUE_FLUSH, null, null)
    }

    override fun onInit(status: Int) {
        if (status == TextToSpeech.SUCCESS) {
            // TTS inicializado com sucesso
            Toast.makeText(this, "Sistema de voz pronto!", Toast.LENGTH_SHORT).show()
        } else {
            Toast.makeText(this, "Erro no sistema de voz", Toast.LENGTH_SHORT).show()
        }
    }

    override fun onDestroy() {
        // Liberar recursos do TTS
        tts.stop()
        tts.shutdown()
        super.onDestroy()
    }
}
